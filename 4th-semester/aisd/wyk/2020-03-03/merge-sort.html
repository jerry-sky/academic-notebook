if n == 1 then done
else
  B = MergeSort( A[ 1,..., floor(n/2) ] )
  C = MergeSort( A[ floor(n/2), ..., n ] )
  merge(B, C)

Nierekurencyjna wersja

iterativeMergeSort(A, n)
inject(Q, x) - działa jak push na liście Q
eject(Q) - działa jak pop na liście Q

Q = []
for i = 1 to n
  inject(Q, A[i])
while |Q| > 1
    inject(Q, merge(eject(Q), eject(Q)))
return eject(Q)
